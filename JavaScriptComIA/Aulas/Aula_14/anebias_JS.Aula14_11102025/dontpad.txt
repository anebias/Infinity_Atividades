      function handleCadastrarProduto(event) {
        // 1. Previne o Comportamento Padrão
        event.preventDefault();

        // 2. Busca o Formulario
        // const form = document.getElementById("produtos-form");
        const form = event.target;

        // 3. Pega os valores dos inputs
        const produto = {
          nome: form.nome.value,
          categoria: form.categoria.value,
          preco: Number(form.preco.value),
        };

        // 4. Busca a tag "tbody"
        const tableData = document.querySelector("#produtos-table tbody");

        // tableData.innerHTML += `
        //     <tr>
        //         <td>${produto.nome}</td>
        //         <td>${produto.categoria}</td>
        //         <td>${produto.preco.toLocaleString("pt-BR", {
        //           style: "currency",
        //           currency: "BRL",
        //         })}</td>
        //     </tr>
        // `;

        // 5. Cria uma linha "tr"
        const trProduto = document.createElement("tr"); // <tr></tr>

        // 6. Cria as celulas "td" e adiciona na linha
        const tdNome = document.createElement("td"); // <td></td>
        tdNome.textContent = produto.nome;
        trProduto.appendChild(tdNome); // <tr> <td>{produto.nome}</td> </tr>

        const tdCategoria = document.createElement("td");
        tdCategoria.textContent = produto.categoria;
        trProduto.appendChild(tdCategoria);

        const tdPreco = document.createElement("td");
        tdPreco.textContent = produto.preco.toLocaleString("pt-BR", {
          style: "currency",
          currency: "BRL",
        });
        trProduto.appendChild(tdPreco);
 
        // 7. Adiciono a linha na tabela
        tableData.appendChild(trProduto)

        form.reset();
      }

      // Busca o Formulário "form"
      const formProduto = document.getElementById("produtos-form");

      // Adiciona o evento de Submit
      // Exemplo: form.addEventListener('submit', handleCadastrarProduto)
      formProduto.addEventListener("submit", handleCadastrarProduto);
    </script>
